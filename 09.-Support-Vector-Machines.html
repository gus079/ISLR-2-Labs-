sim_data <- tibble(
  x1 = rnorm(40),
  x2 = rnorm(40),
  y = factor(rep(c(-1, 1), 20))
  ) %>% 
  mutate(x1 = ifelse(y == 1, x1 + 1.5, x1),

linear_svm_wf <- workflow() %>%
  add_model(svm_linear_model %>% set_args(cost = tune())) %>%
  add_formula(y ~ .)
